def get_accuracy_score(C):
    lin_svm = sk_svm.SVC(kernel='linear',C=C)
    model_accuracy_score=sk_model_selection.cross_val_score(lin_svm, X, y, cv=4,scoring="accuracy")
    return np.mean(model_accuracy_score)
